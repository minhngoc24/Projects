10   public class Nim {
11       public static void main(String[] args) {
12           Scanner scanner = new Scanner(System.in);
13           int pileA = 5;
14           int pileB = 4;
15           int pileC = 3;
16   
17           int currentPlayer = 1;
18   
19           while (true) {
20               System.out.println("Player " + currentPlayer + ", it's your turn");
21               System.out.println("Here is the board: ");
22               System.out.println("A  B  C");
23               System.out.println(pileA + "  " + pileB + "  " + pileC + "  ");
24   
25               // Get user input for pile
26               System.out.println("Enter the pile: ");
27               String selectedPile = scanner.next().toUpperCase();
28   
29               // If User input is different from A B or C
30               if(!(selectedPile.equals("A") || selectedPile.equals("B") || selectedPile.equals("C"))){
31                   System.out.println("Invalid pile; please re-enter the pile:");
32                   continue;
33               }
34   
35               // Get user input for the number of sticks
36               System.out.println("How many sticks would you like to remove from pile " + selectedPile + "?");
37               int removeStick = scanner.nextInt();
38   
39               // If the number of sticks to remove is not positive or higher than the current stick
40               if (selectedPile.equals("A")) {
41                   if (removeStick > pileA || removeStick <= 0) {
42                       System.out.println("Invalid number of sticks; please re-enter number:");
43                       continue;
44                   }
45               } else if (selectedPile.equals("B")) {
46                   if (removeStick > pileB || removeStick <= 0) {
47                       System.out.println("Invalid number of sticks; please re-enter number:");
48                       continue;
49                   }
50               } else if (selectedPile.equals("C")) {
51                   if (removeStick > pileC || removeStick <= 0) {
52                       System.out.println("Invalid number of sticks; please re-enter number:");
53                       continue;
54                   }
55               }
56   
57               // Update the current stick:
58               if (selectedPile.equals("A")) {
59                   pileA -= removeStick;
60               } else if (selectedPile.equals("B")) {
61                   pileB -= removeStick;
62               } else if (selectedPile.equals("C")) {
63                   pileC -= removeStick;
64               }
65   
66               // End game
67               if (pileA + pileB + pileC == 0) {
68                   int winner = 3 - currentPlayer;
69                   System.out.println("The winner is player " + winner + "!");
70                   break;
71               }
72   
73               // Switch player
74               currentPlayer = 3 - currentPlayer;
75           }
76   
77           // Close the scanner
78           scanner.close();
79       }
80   }
